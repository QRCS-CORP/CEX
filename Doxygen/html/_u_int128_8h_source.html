393  inline static void Transpose(UInt128 &X0, UInt128 &X1, UInt128 &X2, UInt128 &X3)
394  {
395  __m128i T0 = _mm_unpacklo_epi32(X0.xmm, X1.xmm);
396  __m128i T1 = _mm_unpacklo_epi32(X2.xmm, X3.xmm);
397  __m128i T2 = _mm_unpackhi_epi32(X0.xmm, X1.xmm);
398  __m128i T3 = _mm_unpackhi_epi32(X2.xmm, X3.xmm);
399  X0.xmm = _mm_unpacklo_epi64(T0, T1);
400  X1.xmm = _mm_unpackhi_epi64(T0, T1);
401  X2.xmm = _mm_unpacklo_epi64(T2, T3);
402  X3.xmm = _mm_unpackhi_epi64(T2, T3);
403  }
