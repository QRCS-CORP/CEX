386 
387  switch (blockBits)
388  {
389  case 128:
390  BC = 4;
391  break;
392  case 192:
393  BC = 6;
394  break;
395  case 256:
396  BC = 8;
397  break;
398  default:
399  return (-2);
400  }
401 
402  switch (keyBits >= blockBits ? keyBits : blockBits)
403  {
404  case 128:

406  break;
407  case 192:
408  ROUNDS = 12;
409  break;

411  ROUNDS = 14;
412  break;

557  ShiftRow(a, 1, BC);
