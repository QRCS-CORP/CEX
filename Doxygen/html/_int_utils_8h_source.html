407  // Different computer architectures store data using different byte orders. "Big-endian"
408  // means the most significant byte is on the left end of a word. "Little-endian" means the
409  // most significant byte is on the right end of a word. i.e.:
410  // BE: uint(block[3]) | (uint(block[2]) << 8) | (uint(block[1]) << 16) | (uint(block[0]) << 24)
411  // LE: uint(block[0]) | (uint(block[1]) << 8) | (uint(block[2]) << 16) | (uint(block[3]) << 24)

434  template<typename ArrayA, typename ArrayB>
435  inline static void BlockToBe(const ArrayA &Input, size_t InOffset, ArrayB &Output, size_t OutOffset, size_t Length)
436  {
437  CexAssert(sizeof(ArrayA::value_type) == sizeof(byte), "Input must be a byte array");
438  CexAssert((Input.size() - InOffset) >= Length, "Length is larger than input capacity");
439  CexAssert((Output.size() - OutOffset) * sizeof(ArrayB::value_type) >= Length, "Length is larger than output capacity");
440 
441 #if defined(IS_BIG_ENDIAN)
442  Utility::MemUtils::Copy(Input, InOffset, Output, OutOffset, Length);
443 #else
444  const size_t VARLEN = sizeof(ArrayB::value_type);
445 
446  for (size_t i = 0; i < Length / VARLEN; ++i)
447  {
448  for (size_t j = VARLEN; j > 0; --j)
449  {
450  Output[OutOffset + i] |= static_cast<ArrayB::value_type>(Input[InOffset + (i * VARLEN) + (VARLEN - j)]) << (8 * (j - 1));
451  }
452  }
453 #endif
454  }

722  template<typename Array>
723  inline static void BeIncrease8(Array &Output, const size_t Length)
724  {
725  CexAssert(sizeof(Array::value_type) == sizeof(byte), "Input and Output must be an array of 8bit integers");
726  CexAssert(!std::is_signed<Array::value_type>::value, "Input and Output must be an unsigned integer array");
727 
728  const int CTRLEN = static_cast<int>(Output.size() - 1);
729  uint ctrLen = static_cast<uint>(Length);
730  std::array<byte, sizeof(uint)> ctrInc;
731 
732  std::memcpy(&ctrInc[0], &ctrLen, ctrInc.size());
733  byte carry = 0;
734 
735  for (int i = CTRLEN; i >= 0; --i)
736  {
737  byte odst = Output[i];
738  byte osrc = CTRLEN - i < static_cast<int>(ctrInc.size()) ? static_cast<byte>(ctrInc[CTRLEN - i]) : static_cast<byte>(0);
739  byte ndst = static_cast<byte>(odst + osrc + carry);
740  carry = ndst < odst ? 1 : 0;
741  Output[i] = ndst;
742  }
743  }
